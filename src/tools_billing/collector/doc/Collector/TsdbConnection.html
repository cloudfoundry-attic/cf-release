def unbind
  @logger.warn("Connection to TSDB server dropped.")

  EM.add_timer(1.0) do
    begin
      reconnect(@ip, @port)
    rescue EventMachine::ConnectionError => e
      @logger.warn(e)
      unbind
    end
  end
end
